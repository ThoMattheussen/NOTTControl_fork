#include "stdafx.h"
#include "ArenaCApi.h"
#include "SaveCApi.h"

#define TAB1 "  "
#define TAB2 "    "

// Timer: Long Exposure
//    This example introduces how the timer can be used to control specific features
//    within the camera. In this case, we use the exposure as an example. Most cameras
//    have a maxiumum exposure limit, but this provides the ability to bypass the
//    exposure limit through the use of timer and trigger. First, the timer and trigger
//    settings are configured so that the timer is connected to the trigger, which is
//    set to activate the exposure. As the timer begins, the trigger activates and the
//    device starts to expose an image. Once the timer duration expires, the exposure
//    stops, the device grabs the image, and the image is saved to disk.

// =-=-=-=-=-=-=-=-=-
// =-=- SETTINGS =-=-
// =-=-=-=-=-=-=-=-=-

// system timeout
#define SYSTEM_TIMEOUT 100

// timer settings (in milliseconds)
#define TIMER_DURATION 15000

// filename
#define FILE_NAME "Images/C_Timer_LongExposure/image.jpg"

// maximum buffer length
#define MAX_BUF 1024

// =-=-=-=-=-=-=-=-=-
// =-=- EXAMPLE -=-=-
// =-=-=-=-=-=-=-=-=-

// demonstrates the use of timer to configure exposure time
// (1) configures the timer and trigger
// (2) starts stream
// (3) starts timer, which triggers the exposure
// (4) once timer duration is up, acquire image
// (5) save long exposure image to disk
AC_ERROR ConfigureTimer(acDevice hDevice)
{
        AC_ERROR err = AC_ERR_SUCCESS;
        SC_ERROR saveErr = SC_ERR_SUCCESS;

        // get node map and retrieve initial node values that will be changed in
        // order to return their values at the end of the example
        acNodeMap hNodeMap = NULL;
        char triggerModeInitial[MAX_BUF];
        size_t triggerModeBufLen = MAX_BUF;
        char triggerSourceInitial[MAX_BUF];
        size_t triggerSourceBufLen = MAX_BUF;
        char triggerSelectorInitial[MAX_BUF];
        size_t triggerSelectorBufLen = MAX_BUF;
        char triggerActivationInitial[MAX_BUF];
        size_t triggerActivationBufLen = MAX_BUF;
        char timerSelectorInitial[MAX_BUF];
        size_t timerSelectorBufLen = MAX_BUF;
        char timerTriggerSourceInitial[MAX_BUF];
        size_t timerTriggerSourceBufLen = MAX_BUF;
        double timerDurationInitial;
        char exposureAutoInitial[MAX_BUF];
        size_t exposureAutoBufLen = MAX_BUF;
        char softwareSignalSelectorInitial[MAX_BUF];
        size_t softwareSignalSelectorBufLen = MAX_BUF;

        err = acDeviceGetNodeMap(hDevice, &hNodeMap);
        if (err != AC_ERR_SUCCESS)
                return err;

        err = acNodeMapGetEnumerationValue(hNodeMap, "TriggerSelector", triggerSelectorInitial, &triggerSelectorBufLen) |
                  acNodeMapGetEnumerationValue(hNodeMap, "TriggerMode", triggerModeInitial, &triggerModeBufLen) |
                  acNodeMapGetEnumerationValue(hNodeMap, "TriggerSource", triggerSourceInitial, &triggerSourceBufLen) |
                  acNodeMapGetEnumerationValue(hNodeMap, "TriggerActivation", triggerActivationInitial, &triggerActivationBufLen) |
                  acNodeMapGetEnumerationValue(hNodeMap, "TimerSelector", timerSelectorInitial, &timerSelectorBufLen) |
                  acNodeMapGetEnumerationValue(hNodeMap, "TimerTriggerSource", timerTriggerSourceInitial, &timerTriggerSourceBufLen) |
                  acNodeMapGetFloatValue(hNodeMap, "TimerDuration", &timerDurationInitial) |
                  acNodeMapGetEnumerationValue(hNodeMap, "ExposureAuto", exposureAutoInitial, &exposureAutoBufLen) |
                  acNodeMapGetEnumerationValue(hNodeMap, "SoftwareSignalSelector", softwareSignalSelectorInitial, &softwareSignalSelectorBufLen);
        if (err != AC_ERR_SUCCESS)
                printf("\nWarning: failed to retrieve one or more initial node values.\n");

        // Setting required timer and trigger settings
        //    If any of these do not work, it may be due to the camera type
        //    or camera firmware. Try updating the firmware or using a
        //    different camera.

        // Set trigger selector
        //    Set the trigger selector to Exposure Active. When triggered,
        //    the device will start exposing an image.
        printf("%sSet trigger selector to Exposure Active\n", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "TriggerSelector", "ExposureActive");
        if (err != AC_ERR_SUCCESS)
                return err;

        // Set trigger source
        //    Set the trigger source to Timer 0 Active. This connects
        //    the trigger to the timer.
        printf("%sSet trigger source to Timer 0 Active\n", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "TriggerSource", "Timer0Active");
        if (err != AC_ERR_SUCCESS)
                return err;

        // Set trigger activation
        //    Set the trigger source to Level High. This ensures
        //    that the trigger is activated when the timer is running.
        printf("%sSet triger activation to Level High\n", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "TriggerActivation", "LevelHigh");
        if (err != AC_ERR_SUCCESS)
                return err;

        // Set trigger mode
        //    Set trigger mode to on. This enables trigger mode
        //    on the camera.
        printf("%sSet trigger mode to On\n", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "TriggerMode", "On");
        if (err != AC_ERR_SUCCESS)
                return err;

        // set timer selector
        printf("%sSet timer selector to Timer 0\n", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "TimerSelector", "Timer0");
        if (err != AC_ERR_SUCCESS)
                return err;

        // Set timer trigger source
        //    Set timer trigger source to Software Signal 0. Timer is only
        //    activated through software by calling the SoftwareSignal node.
        printf("%sSet timer trigger source to Software Signal 0\n", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "TimerTriggerSource", "SoftwareSignal0");
        if (err != AC_ERR_SUCCESS)
                return err;

        // Set timer duration
        //    Set the timer duration to user-set duration. This gives user
        //    ability to control the duration of specific camera events
        //    without limitations, such as exposure. The timer duration
        //    node is configured in microseconds.
        printf("%sSet timer duration to %d milliseconds\n", TAB1, TIMER_DURATION);

        // converts TIMER_DURATION into microseconds
        float timerDuration = TIMER_DURATION * 1000;
        err = acNodeMapSetFloatValue(hNodeMap, "TimerDuration", timerDuration);
        if (err != AC_ERR_SUCCESS)
                return err;

        // Set exposure auto
        //    Set exposure auto to off. Ensures the exposure time is only
        //    manipulated by user.
        printf("%sSet exposure auto to Off\n", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "ExposureAuto", "Off");
        if (err != AC_ERR_SUCCESS)
                printf("%sExposure auto not a feature available\n", TAB2);

        // Set software signal selector
        //    Set software signal selector to Software Signal 0. Connects software
        //    signal to timer trigger.
        printf("%sSet software signal selector to Software Signal 0\n", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "SoftwareSignalSelector", "SoftwareSignal0");
        if (err != AC_ERR_SUCCESS)
                return err;

        // get stream node map
        acNodeMap hTLStreamNodeMap = NULL;

        err = acDeviceGetTLStreamNodeMap(hDevice, &hTLStreamNodeMap);
        if (err != AC_ERR_SUCCESS)
                return err;

        // enable stream auto negotiate packet size
        err = acNodeMapSetBooleanValue(hTLStreamNodeMap, "StreamAutoNegotiatePacketSize", true);
        if (err != AC_ERR_SUCCESS)
                return err;

        // enable stream packet resend
        err = acNodeMapSetBooleanValue(hTLStreamNodeMap, "StreamPacketResendEnable", true);
        if (err != AC_ERR_SUCCESS)
                return err;

        // start stream
        printf("%sStart stream\n", TAB1);

        err = acDeviceStartStream(hDevice);
        if (err != AC_ERR_SUCCESS)
                return err;

        // Trigger Armed
        //    Continually check until trigger is armed. Once the trigger is armed, it
        //    is ready to be executed.
        printf("%sWait until trigger is armed\n", TAB2);

        bool8_t triggerArmed = false;

        do
        {
                acNodeMapGetBooleanValue(hNodeMap, "TriggerArmed", &triggerArmed);
        } while (triggerArmed == false);

        // Start timer
        //    This is done through a software signal. This also triggers the
        //    exposure active trigger, which starts to expose the image.
        printf("%sStart timer for %f seconds\n", TAB2, TIMER_DURATION / 1e3);

        err = acNodeMapExecute(hNodeMap, "SoftwareSignalPulse");
        if (err != AC_ERR_SUCCESS)
                return err;

        // Acquire image
        //    This waits for the length of the timer duration to get an image.
        //    If an image buffer is not ready by the timeout duration, it fails.
        //    The result should be an image exposed to the duration of the timer.
        printf("%sGet image\n", TAB2);
        acBuffer hBuffer = NULL;

        uint64_t timeout = (uint64_t)timerDuration;
        err = acDeviceGetBuffer(hDevice, timeout, &hBuffer);
        if (err != AC_ERR_SUCCESS)
                return err;

        // prepare image parameters
        printf("%sSave image to ", TAB2);

        size_t width = 0;
        size_t height = 0;
        size_t bpp = 0;

        err = acImageGetWidth(hBuffer, &width) |
                  acImageGetHeight(hBuffer, &height) |
                  acImageGetBitsPerPixel(hBuffer, &bpp);
        if (err != AC_ERR_SUCCESS)
                return err;

        // prepare image writer
        saveWriter hWriter = NULL;

        saveErr = saveWriterCreate(width, height, bpp, &hWriter) |
                          saveWriterSetFileNamePattern(hWriter, FILE_NAME);
        if (saveErr != SC_ERR_SUCCESS)
                return saveErr;

        // set image writer to jpeg
        saveErr = saveWriterSetJpeg(hWriter);
        if (saveErr != SC_ERR_SUCCESS)
                return saveErr;

        // get image
        uint8_t* pData = NULL;

        err = acImageGetData(hBuffer, &pData);
        if (err != AC_ERR_SUCCESS)
                return err;

        // save image
        saveErr = saveWriterSave(hWriter, pData);
        if (saveErr != SC_ERR_SUCCESS)
                return saveErr;

        printf("%s\n", FILE_NAME);

        // destroy image writer
        saveErr = saveWriterDestroy(hWriter);
        if (saveErr != SC_ERR_SUCCESS)
                return saveErr;

        // requeue image buffer
        err = acDeviceRequeueBuffer(hDevice, hBuffer);
        if (err != AC_ERR_SUCCESS)
                return err;

        // Stop the stream
        printf("%sStop stream\n", TAB1);

        err = acDeviceStopStream(hDevice);
        if (err != AC_ERR_SUCCESS)
                return err;

        // return nodes to their initial values
        err = acNodeMapSetEnumerationValue(hNodeMap, "ExposureAuto", exposureAutoInitial);
        if (err != AC_ERR_SUCCESS)
                printf("%sExposure auto not a feature available", TAB1);

        err = acNodeMapSetEnumerationValue(hNodeMap, "TriggerSource", triggerSourceInitial) |
                  acNodeMapSetEnumerationValue(hNodeMap, "TriggerMode", triggerModeInitial) |
                  acNodeMapSetEnumerationValue(hNodeMap, "TriggerSelector", triggerSelectorInitial) |
                  acNodeMapSetEnumerationValue(hNodeMap, "TriggerActivation", triggerActivationInitial) |
                  acNodeMapSetEnumerationValue(hNodeMap, "TimerSelector", timerSelectorInitial) |
                  acNodeMapSetEnumerationValue(hNodeMap, "TimerTriggerSource", timerTriggerSourceInitial) |
                  acNodeMapSetFloatValue(hNodeMap, "TimerDuration", timerDurationInitial) |
                  acNodeMapSetEnumerationValue(hNodeMap, "SoftwareSignalSelector", softwareSignalSelectorInitial);
        if (err != AC_ERR_SUCCESS)
                printf("\nWarning: failed to set one or more node values back to its initial value.\n");

        return err;
}

AC_ERROR SelectDevice(acSystem hSystem, size_t* pNumDevices, size_t* pSelection)
{
        AC_ERROR err = AC_ERR_SUCCESS;

        if (*pNumDevices == 1)
        {
                printf(TAB1 "Only one device detected, automatically selecting this device.\n");
                *pSelection = 0;
                return AC_ERR_SUCCESS;
        }

        printf(TAB1 "Select device:\n");
        for (size_t i = 0; i < *pNumDevices; i++)
        {
                // get device model
                char pDeviceModel[MAX_BUF];
                size_t pDeviceModelLen = MAX_BUF;
                err = acSystemGetDeviceModel(hSystem, i, pDeviceModel, &pDeviceModelLen);
                if (err != AC_ERR_SUCCESS)
                        return err;

                // get device serial
                char pDeviceSerial[MAX_BUF];
                size_t pDeviceSerialLen = MAX_BUF;
                err = acSystemGetDeviceSerial(hSystem, i, pDeviceSerial, &pDeviceSerialLen);
                if (err != AC_ERR_SUCCESS)
                        return err;

                // get device IP address
                char pIpAddressStr[MAX_BUF];
                size_t pIpAddressStrBufLen = MAX_BUF;
                err = acSystemGetDeviceIpAddressStr(hSystem, i, pIpAddressStr, &pIpAddressStrBufLen);
                if (err != AC_ERR_SUCCESS)
                        return err;

                printf(TAB2 "%zu. %s%s%s%s%s\n", i + 1, pDeviceModel, TAB1, pDeviceSerial, TAB1, pIpAddressStr);
        }

        do
        {
                printf(TAB1 "Make selection (1-%zu): ", *pNumDevices);

                if (scanf_s("%zu", pSelection) != 1)
                {
                        while (getchar() != '\n')
                                ;
                        printf(TAB1 "Invalid input. Please enter a number.\n");
                        continue;
                }

                if (*pSelection <= 0 || *pSelection > *pNumDevices)
                {
                        printf(TAB1 "Invalid device selected. Please select a device in the range (1-%zu).\n", *pNumDevices);
                }
        } while (*pSelection <= 0 || *pSelection > *pNumDevices);

        *pSelection -= 1;
        return AC_ERR_SUCCESS;
}

// error buffer length
#define ERR_BUF 512

#define CHECK_RETURN                                  \
        if (err != AC_ERR_SUCCESS)                        \
        {                                                 \
                char pMessageBuf[ERR_BUF];                    \
                size_t pBufLen = ERR_BUF;                     \
                acGetLastErrorMessage(pMessageBuf, &pBufLen); \
                printf("\nError: %s", pMessageBuf);           \
                printf("\n\nPress enter to complete\n");      \
                getchar();                                    \
                return -1;                                    \
        }

int main()
{
        printf("C_Timer_LongExposure\n");
        AC_ERROR err = AC_ERR_SUCCESS;

        // prepare example
        acSystem hSystem = NULL;
        err = acOpenSystem(&hSystem);
        CHECK_RETURN;
        err = acSystemUpdateDevices(hSystem, SYSTEM_TIMEOUT);
        CHECK_RETURN;
        size_t numDevices = 0;
        err = acSystemGetNumDevices(hSystem, &numDevices);
        CHECK_RETURN;
        if (numDevices == 0)
        {
                printf("\nNo camera connected\nPress enter to complete\n");
                getchar();
                return -1;
        }
        acDevice hDevice = NULL;
        size_t selection = 0;
        err = SelectDevice(hSystem, &numDevices, &selection);
        CHECK_RETURN;
        err = acSystemCreateDevice(hSystem, selection, &hDevice);
        CHECK_RETURN;

        // run example
        printf("Commence example\n\n");
        err = ConfigureTimer(hDevice);
        CHECK_RETURN;
        printf("\nExample complete\n");

        // clean up example
        err = acSystemDestroyDevice(hSystem, hDevice);
        CHECK_RETURN;
        err = acCloseSystem(hSystem);
        CHECK_RETURN;

        printf("Press enter to complete\n");
        while (getchar() != '\n') {};
        getchar();
        return -1;
}
