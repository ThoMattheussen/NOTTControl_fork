#include "stdafx.h"
#include "ArenaApi.h"
#include "SaveApi.h"

#define TAB1 "  "
#define TAB2 "    "

// Multipart Save: Introduction
//        Mutlipart mode combines smaller image segments (parts) outputted by the camera to
//    form a complete image.
//    This example covers enabling multipart mode, setting the number of parts to coalesce
//    into a single image and saving the image for visualization.
//    NOTE: This functionality is available for linescan cameras only

// =-=-=-=-=-=-=-=-=-
// =-=- SETTINGS =-=-
// =-=-=-=-=-=-=-=-=-

// Number of Parts
//        The number of parts to be coalesced together to form a single image.
//    NOTE: NUM_PARTS should not exceed the maximum allowed, which depends on the
//    max height and the current height
#define NUM_PARTS 3

// Pixel format
#define PIXEL_FORMAT BGR8

// File name
#define FILE_NAME "Images/Cpp_Multipart_Save/multipart_image.png"

// =-=-=-=-=-=-=-=-=-
// =-=- EXAMPLE -=-=-
// =-=-=-=-=-=-=-=-=-

void SaveImage(Arena::IImage* pImage, const char* filename)
{
        // Convert image
        std::cout << TAB2 << "Convert image to " << GetPixelFormatName(PIXEL_FORMAT) << "\n";

        auto pConverted = Arena::ImageFactory::Convert(
                pImage,
                PIXEL_FORMAT);

        // Prepare image parameters
        std::cout << TAB2 << "Prepare image parameters\n";

        Save::ImageParams params(
                pConverted->GetWidth(),
                pConverted->GetHeight(),
                pConverted->GetBitsPerPixel());

        // Prepare image writer
        std::cout << TAB2 << "Prepare image writer\n";

        Save::ImageWriter writer(
                params,
                filename);

        // Save image
        std::cout << TAB2 << "Save image to ";

        writer << pConverted->GetData();

        std::cout << writer.GetLastFileName() << "\n";

        // destroy converted image
        Arena::ImageFactory::Destroy(pConverted);
}

int64_t calculateMaxNumParts(Arena::IDevice* pDevice)
{
        GenApi::CIntegerPtr pHeight = pDevice->GetNodeMap()->GetNode("Height");

        int64_t maxHeight = pHeight->GetMax();
        int64_t currentHeight = pHeight->GetValue();

        // Calculate the maximum number of image parts to be stacked vertically
        // to form a complete image. This is determined by dividing the maximum
        // allowed height by the current height.
        int64_t maxNumParts = static_cast<int64_t>(std::floor(maxHeight / currentHeight));

        return maxNumParts;
}

// demonstrates acquiring and saving multipart image
// (1) ensures multipart is implemented for your camera
// (2) enables multipart
// (3) ensures number of parts is less than the maximum allowed
// (4) sets number of parts
// (5) starts the stream
// (6) acquires one image
// (7) saves the image
// (8) requeues the buffer
// (9) stops the stream

void AcquireAndSaveMultipartImage(Arena::IDevice* pDevice)
{
        // Multipart is available for linescan cameras only
        if (!GenApi::IsAvailable(pDevice->GetTLDeviceNodeMap()->GetNode("StreamMultipartEnable"))) {
                std::cout << "Multipart is only available for linescan cameras. Please try with a linescan camera.\n";
                return;
        }

        std::cout << TAB1 << "Set Stream Multipart Enable to 'true'" << "\n";

        // When StreamMultipartEnable is set to true, GenTL will accumulate
        // NUM_PARTS number of images to coalesce into a single complete image.
        Arena::SetNodeValue<bool>(
                pDevice->GetTLDeviceNodeMap(),
                "StreamMultipartEnable",
                true);

        // maxiumum number of parts depends on the current camera height
        // reducing the height will allow more parts
        int64_t maxNumParts = calculateMaxNumParts(pDevice);
        int64_t numParts = NUM_PARTS < maxNumParts ? NUM_PARTS : maxNumParts;

        // set number of parts to NUM_PARTS or maxiumum allowed
        std::cout << TAB2 << "Set Stream Multipart NumParts to " << numParts << "\n";
        Arena::SetNodeValue<int64_t>(
                pDevice->GetTLStreamNodeMap(),
                "StreamMultipartNumParts",
                numParts);

        // packet resend should be enabled after multipart is enabled
        // because packet resend enable is per stream type
        Arena::SetNodeValue<bool>(pDevice->GetTLStreamNodeMap(), "StreamPacketResendEnable", true);

        // start stream
        std::cout << TAB1 << "Start stream\n";
        pDevice->StartStream();

        std::cout << TAB2 << "Get one image\n";

        // The retrieved image will have NUM_PARTS parts combined into one image
        Arena::IImage* pImage = pDevice->GetImage(2000);

        SaveImage(pImage, FILE_NAME);

        // clean up example
        pDevice->RequeueBuffer(pImage);

        std::cout << TAB1 << "Stop stream\n";

        // stop stream
        pDevice->StopStream();
}

Arena::DeviceInfo SelectDevice(std::vector<Arena::DeviceInfo>& deviceInfos)
{
        if (deviceInfos.size() == 1)
        {
                std::cout << "\n"
                                  << TAB1 << "Only one device detected: " << deviceInfos[0].ModelName() << TAB1 << deviceInfos[0].SerialNumber() << TAB1 << deviceInfos[0].IpAddressStr() << ".\n";
                std::cout << TAB1 << "Automatically selecting this device.\n";
                return deviceInfos[0];
        }

        std::cout << "\nSelect device:\n";
        for (size_t i = 0; i < deviceInfos.size(); i++)
        {
                std::cout << TAB1 << i + 1 << ". " << deviceInfos[i].ModelName() << TAB1 << deviceInfos[i].SerialNumber() << TAB1 << deviceInfos[i].IpAddressStr() << "\n";
        }
        size_t selection = 0;

        do
        {
                std::cout << TAB1 << "Make selection (1-" << deviceInfos.size() << "): ";
                std::cin >> selection;

                if (std::cin.fail())
                {
                        std::cin.clear();
                        while (std::cin.get() != '\n')
                                ;
                        std::cout << TAB1 << "Invalid input. Please enter a number.\n";
                }
                else if (selection <= 0 || selection > deviceInfos.size())
                {
                        std::cout << TAB1 << "Invalid device selected. Please select a device in the range (1-" << deviceInfos.size() << ").\n";
                }

        } while (selection <= 0 || selection > deviceInfos.size());

        return deviceInfos[selection - 1];
}

int main()
{
        // flag to track when an exception has been thrown
        bool exceptionThrown = false;

        std::cout << "Cpp_Multipart_Save";

        try
        {
                // prepare example
                Arena::ISystem* pSystem = Arena::OpenSystem();
                pSystem->UpdateDevices(100);
                std::vector<Arena::DeviceInfo> devices = pSystem->GetDevices();
                if (devices.size() == 0)
                {
                        std::cout << "\nNo camera connected\nPress enter to complete\n";
                        std::getchar();
                        return 0;
                }
                Arena::DeviceInfo selectedDevice = SelectDevice(devices);
                Arena::IDevice* pDevice = pSystem->CreateDevice(selectedDevice);

                // enable stream auto negotiate packet size
                Arena::SetNodeValue<bool>(pDevice->GetTLStreamNodeMap(), "StreamAutoNegotiatePacketSize", true);

                std::cout << "Commence example\n\n";
                AcquireAndSaveMultipartImage(pDevice);
                std::cout << "\nExample complete\n";

                pSystem->DestroyDevice(pDevice);
                Arena::CloseSystem(pSystem);
        }
        catch (GenICam::GenericException& ge)
        {
                std::cout << "\nGenICam exception thrown: " << ge.what() << "\n";
                exceptionThrown = true;
        }
        catch (std::exception& ex)
        {
                std::cout << "\nStandard exception thrown: " << ex.what() << "\n";
                exceptionThrown = true;
        }
        catch (...)
        {
                std::cout << "\nUnexpected exception thrown\n";
                exceptionThrown = true;
        }

        std::cout << "Press enter to complete\n";
        std::cin.ignore();
        std::getchar();

        if (exceptionThrown)
                return -1;
        else
                return 0;
}
